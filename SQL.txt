--create table
--Set of Rules->Constraints
--Primary key,Foreign key,Check,default,unique,not null
create table IBMTrainees
(
TID int Primary key,
TName varchar(25) Unique not null,
Designation varchar(25) default 'Trainee',
Age int check(Age<24)
)
